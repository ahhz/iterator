Motivating Example

The following is an example of a program iterating over three ranges in parallel.

#include <blink\iterator\zip_range.h>

#include <functional>
#include <iostream>
#include <string>
#include <vector>

int main()
{
  std::vector<int> ints = { 1, 2, 3, 4 };
  std::vector<std::string> roman = { "I", "II", "III", "IV" };
  std::vector<std::string> strs = { "one", "two", "three", "four" };
  
  // use std::ref to avoid copying the range
  auto zip_range = blink::iterator::make_zip_range(std::ref(ints), std::ref(roman), strs);
  
  for (auto&& i : zip_range)
  {
    std::cout << std::get<0>(i) << " " << std::get<1>(i) << " " << std::get<2>(i) << std::endl;
  }
  return 0;
}

The output is:

1 I one
2 II two
3 III three
4 IV four
